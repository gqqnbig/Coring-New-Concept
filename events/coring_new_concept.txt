add_namespace = cor_sta

#starting integration
country_event = {
	id = cor_sta.1
	fire_only_once = no
	mean_time_to_happen = { days = 1 }
	hidden = no
	
	title = cor_sta.1.t
	desc = cor_sta.1.d
	
	trigger = {
		any_state = {
			can_integrate = yes
		}
	}
	
	option = { #fast integration
		trigger = {
			AND = {
				has_political_power > 29
				OR = {
					is_ai = no
					has_political_power > 299
				}
			}
		}
		name = cor_sta.1.a
		add_political_power = -15
		random_state = {
			limit = {
				can_integrate = yes
			}
			clr_state_flag = fast_coring #no claim, no coring
			clr_state_flag = normal_coring
			clr_state_flag = slow_coring
			set_state_flag = fast_integration
			custom_effect_tooltip = cor_sta.1.a.t
		}
	}
	
	option = { #normal integration
		trigger = {
			AND = {
				has_political_power > 9
				OR = {
					is_ai = no
					has_political_power > 159
				}
			}
		}
		name = cor_sta.1.b
		add_political_power = -5
		random_state = {
			limit = {
				can_integrate = yes
			}
			clr_state_flag = fast_coring #no claim, no coring
			clr_state_flag = normal_coring
			clr_state_flag = slow_coring
			set_state_flag = normal_integration
			custom_effect_tooltip = cor_sta.1.b.t
		}
	}
	
	option = { #slow integration
		name = cor_sta.1.c
		random_state = {
			limit = {
				can_integrate = yes
			}
			set_state_flag = slow_integration
			clr_state_flag = fast_coring #no claim, no coring
			clr_state_flag = normal_coring
			clr_state_flag = slow_coring
			custom_effect_tooltip = cor_sta.1.c.t
		}
	}
}

country_event = {
	id = cor_sta.2
	fire_only_once = no
	mean_time_to_happen = { days = 1 }
	hidden = no
	
	title = cor_sta.2.t
	desc = cor_sta.2.d
	
	trigger = {
		any_state = {
			is_integrated = yes
		}
	}
	
	option = {
		name = cor_sta.2.a
		every_state = {
			limit = {
				is_integrated = yes
			}
			add_state_modifier = { 
				modifier = { 
					local_non_core_manpower = 0.18
					#resistance_tick_auto = -2 doesn't work
					#resistance_tick = -.25
					local_building_slots_factor = 0.25
				} 
			}
			add_claim_by = FROM
			set_state_flag = int_bonus
			clr_state_flag = fast_integration
			clr_state_flag = normal_integration
			clr_state_flag = slow_integration
			clr_state_flag = fast_coring #no claim previous to this event, no coring
			clr_state_flag = normal_coring
			clr_state_flag = slow_coring
		}
	}
}

#coring
country_event = {
	id = cor_sta.7
	fire_only_once = no
	mean_time_to_happen = { days = 1 }
	hidden = no
	
	title = cor_sta.6.t
	desc = cor_sta.6.d
	
	trigger = {
		any_state = {
			AND = {
				is_owned_by = FROM
				is_claimed_by = FROM
				NOT = { is_core_of = FROM }
				OR = {
					has_state_flag = {
						flag = fast_coring
						inttime*days >80
					}
					has_state_flag = {
						flag = normal_coring
						inttime*days >170
					}
					has_state_flag = {
						flag = slow_coring
						inttime*days >350
					}
				}
			}
		}
	}
	
	option = {
		name = cor_sta.7.a
		every_state = {
			limit = {
				AND = {
					is_owned_by = FROM
					is_claimed_by = FROM
					NOT = { is_core_of = FROM }
					OR = {
						has_state_flag = {
							flag = fast_coring
							inttime*days >80
						}
						has_state_flag = {
							flag = normal_coring
							inttime*days >170
						}
						has_state_flag = {
							flag = slow_coring
							inttime*days >350
						}
					}
				}
			}
			add_core_of = FROM
			hidden_effect = {
				add_state_modifier = { 
					modifier = { 
						local_non_core_manpower = -0.18
						#resistance_tick_auto = 2
						local_building_slots_factor = 0.25
					} 
				}
		
			}
			clr_state_flag = fast_coring
			clr_state_flag = normal_coring
			clr_state_flag = slow_coring
			clr_state_flag = int_bonus
			clr_state_flag = fast_integration
			clr_state_flag = normal_integration
			clr_state_flag = slow_integration
		}
	}
}
